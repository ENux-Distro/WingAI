Here's a clean Bash fastfetch clone:

#!/bin/bash
OS=$(grep PRETTY_NAME /etc/os-release | cut -d= -f2 | tr -d '"')
KERNEL=$(uname -r)
CPU=$(grep "model name" /proc/cpuinfo | head -1 | sed 's/.*: //')
MEM=$(free -h | awk '/Mem:/{print $3"/"$2}')

echo -e "\e[34mOS:\e[0m $OS"
echo -e "\e[34mKernel:\e[0m $KERNEL"
echo -e "\e[34mCPU:\e[0m $CPU"
echo -e "\e[34mMemory:\e[0m $MEM"
